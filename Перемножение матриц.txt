#include <iostream>
using namespace std;

int main() {
    int rows1, cols1, rows2, cols2;
    
    // Ввод размеров первой матрицы
    cout << "Введите количество строк и столбцов первой матрицы: ";
    cin >> rows1 >> cols1;
    
    // Ввод размеров второй матрицы
    cout << "Введите количество строк и столбцов второй матрицы: ";
    cin >> rows2 >> cols2;
    
    // Проверка возможности умножения
    if (cols1 != rows2) {
        cout << "Ошибка: Умножение невозможно!" << endl;
        cout << "Количество столбцов первой матрицы должно равняться количеству строк второй матрицы." << endl;
        return 1;
    }
    
    // Создание матриц
    int matrix1[rows1][cols1];
    int matrix2[rows2][cols2];
    int result[rows1][cols2];
    
    // Ввод первой матрицы
    cout << "Введите элементы первой матрицы:" << endl;
    for (int i = 0; i < rows1; i++) {
        for (int j = 0; j < cols1; j++) {
            cin >> matrix1[i][j];
        }
    }
    
    // Ввод второй матрицы
    cout << "Введите элементы второй матрицы:" << endl;
    for (int i = 0; i < rows2; i++) {
        for (int j = 0; j < cols2; j++) {
            cin >> matrix2[i][j];
        }
    }
    
    // Инициализация результирующей матрицы нулями
    for (int i = 0; i < rows1; i++) {
        for (int j = 0; j < cols2; j++) {
            result[i][j] = 0;
        }
    }
    
    // Умножение матриц
    for (int i = 0; i < rows1; i++) {
        for (int j = 0; j < cols2; j++) {
            for (int k = 0; k < cols1; k++) {
                result[i][j] += matrix1[i][k] * matrix2[k][j];
            }
        }
    }
    
    // Вывод результата
    cout << "Результат умножения матриц:" << endl;
    for (int i = 0; i < rows1; i++) {
        for (int j = 0; j < cols2; j++) {
            cout << result[i][j] << " ";
        }
        cout << endl;
    }
    
    return 0;
}
